const video = document.getElementById('v'); function startSession() { document.getElementById('start-btn').style.display = 'none'; document.getElementById('loader-container').style.display = 'flex'; // Wait until video can actually play video.oncanplay = () => { document.getElementById('loader-container').style.display = 'none'; document.getElementById('video-container').style.display = 'block'; video.play().catch(e => console.log("Playback error:", e)); setTimeout(enableSecurity, 2000); }; } function terminate() { video.pause(); document.getElementById('main-content').style.display = 'none'; document.getElementById('terminated').style.display = 'block'; } function enableSecurity() { // Only terminate if user leaves tab window.addEventListener('blur', terminate); // MUCH softer mouse detection let moveCount = 0; window.addEventListener('mousemove', () => { moveCount++; if (moveCount > 50) terminate(); // allows normal movement }); }
